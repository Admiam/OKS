   public static String najdiZnakFakulty(String fakulta) {
     fakulta = fakulta.toUpperCase();
     for (String[] fakulty : PLATNE_FAKULTY) {
       if (fakulty[0].equals(fakulta) == true) {
         return fakulty[1];
       }     
     }
     return ZNAK_CHYBY;
   }
